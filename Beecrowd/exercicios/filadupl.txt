class Celula {
    int elemento;
    Celula prox;
    Celula ant;

    Celula(int elemento) {
        this.elemento = elemento;
        this.prox = null;
        this.ant = null;
    }
}

class FilaDupla {
    Celula primeiro, ultimo;

    FilaDupla() {
        primeiro = ultimo = null;
    }

    void inserirInicio(int elemento) {
        Celula nova = new Celula(elemento);
        if (primeiro == null) {
            primeiro = ultimo = nova;
        } else {
            nova.prox = primeiro;
            primeiro.ant = nova;
            primeiro = nova;
        }
    }

    void inserirFim(int elemento) {
        Celula nova = new Celula(elemento);
        if (ultimo == null) {
            primeiro = ultimo = nova;
        } else {
            ultimo.prox = nova;
            nova.ant = ultimo;
            ultimo = nova;
        }
    }

    int removerInicio() {
        if (primeiro == null) {
            System.out.println("Erro: fila vazia!");
            return -1;
        } else {
            int removido = primeiro.elemento;
            primeiro = primeiro.prox;
            if (primeiro == null)
                ultimo = null;
            else
                primeiro.ant = null;
            return removido;
        }
    }

    int removerFim() {
        if (ultimo == null) {
            System.out.println("Erro: fila vazia!");
            return -1;
        } else {
            int removido = ultimo.elemento;
            ultimo = ultimo.ant;
            if (ultimo == null)
                primeiro = null;
            else
                ultimo.prox = null;
            return removido;
        }
    }

    boolean buscar(int elemento) {
        Celula atual = primeiro;
        while (atual != null) {
            if (atual.elemento == elemento) {
                return true;
            }
            atual = atual.prox;
        }
        return false;
    }

    void mostrar() {
        if (primeiro == null) {
            System.out.println("Fila vazia");
        } else {
            System.out.print("Fila: ");
            Celula atual = primeiro;
            while (atual != null) {
                System.out.print(atual.elemento + " ");
                atual = atual.prox;
            }
            System.out.println();
        }
    }

    void mostrarReverso() {
        if (ultimo == null) {
            System.out.println("Fila vazia");
        } else {
            System.out.print("Fila reversa: ");
            Celula atual = ultimo;
            while (atual != null) {
                System.out.print(atual.elemento + " ");
                atual = atual.ant;
            }
            System.out.println();
        }
    }
}
